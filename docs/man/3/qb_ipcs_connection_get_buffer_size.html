    NAME

    qb_ipcs_connection_get_buffer_size - Retrieve the connection ipc buffer size.

    SYNOPSIS

    #include <qb/qbipcs.h>

int32_t qb_ipcs_connection_get_buffer_size(

    
    qb_ipcs_connection_t   *conn   /* connection instance */
);

    PARAMS

    conn connection instance

    DESCRIPTION

    This reflects the largest size msg that can be sent or received.

    RETURN VALUE

    msg size in bytes, negative value on error.

    NAME

    struct nvme_eom_lane_desc - EOM Lane Descriptor

    SYNOPSIS

    struct nvme_eom_lane_desc {
    __u8 rsvd0;
    __u8 mstatus;
    __u8 lane;
    __u8 eye;
    __le16 top;
    __le16 bottom;
    __le16 left;
    __le16 right;
    __le16 nrows;
    __le16 ncols;
    __le16 edlen;
    __u8 rsvd18[14];
    __u8 eye_desc[];
    };

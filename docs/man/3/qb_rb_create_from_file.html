    NAME

    qb_rb_create_from_file - Load the saved ring buffer from file into temporary memory.

    SYNOPSIS

    #include <qb/qbrb.h>

qb_ringbuffer_t * qb_rb_create_from_file(

    
    int32_t   fd,     /* file with saved ringbuffer data. */

    
    uint32_t  flags   /* same flags as passed into qb_rb_open()*/
);

    PARAMS

    fd file with saved ringbuffer data.

    flags same flags as passed into qb_rb_open()

    DESCRIPTION

    RETURN VALUE

    new ringbuffer instance

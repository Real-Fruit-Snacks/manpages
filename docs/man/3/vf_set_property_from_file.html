    vf_set_property_from_file - vf_set_property_from_file()

    SYNOPSIS

    extern bool_t vf_set_property_from_file
    (
    VF_PROP_T *p_prop,
    vf_encoding_t encoding,
    const char *p_filename
    );

    PARAMETERS

    VF_PROP_T *p_prop
    The property.
    vf_encoding_t encoding
    Encoding to use.
    const char *p_filename
